Android.mk文件

  • Android
  • JNI
  • NDK
  • Android.mk是一个GNU makefile片段，编译器会解析其一次或多次。

最简单的Android.mk

# Android.mk必须由定义LOCAL_PATH变量开始
# my-dir是编译系统提供的宏，返回当前文件所在文件夹路径
LOCAL_PATH := $(call my-dir)

# 重置几乎所有的LOCAL_XXX变量。必须在每个module描述前声明该句。
include $(CLEAR_VARS)

# 指定生成的文件名。
LOCAL_MODULE := hello-jni

# 指定源文件。（多个源文件时，空格隔开）
LOCAL_SRC_FILES := hello-jni.c

# 指定编译行为。这里是生成动态库.so
include $(BUILD_SHARED_LIBRARY)

宏和变量

如果需要定义自己的变量，可以考虑使用MY_开头，而不是LOCAL_、PRIVATE_、NDK_、APP，因为编译系统内部有使用这些开头的变量。

  • NDK定义的变量
# CLEAR_VARS指向特定的GNU makefile，以可重置几乎所有的LOCAL_XXX变量，如LOCAL_MODULE,LOCAL_SRC_FILES, LOCAL_STATIC_LIBRARIES但不包括LOCAL_PATH。
# 由于编译系统使用单一的GUN Make解析所有makefile文件，所有的变量都是全局变量，因此，每个module必须重定义变量。
include $(CLEAR_VARS)

# 生成动态库.so，至少需要定义LOCAL_MODULE和LOCAL_SRC_FILES。
# BUILD_SHARED_LIBRARY指向特定的GUN Makefile，用于收集定义的所有LOCAL_XXX信息（从离该语句最近的include开始收集）
include $(BUILD_SHARED_LIBRARY)

# 生成静态库.a，编译系统不会将静态库复制到项目或软件包中，但可以用静态库生成动态库。
include $(BUILD_STATIC_LIBRARY)

# 与BUILD_SHARED_LIBRARY、BUILD_STATIC_LIBRARY不同，LOCAL_SRC_FILES需要  
# 指向已编译好的动态库。在其他module中可以指定LOCAL_PREBUILTS来使用这里生成的动态库。
include $(PREBUILT_SHARED_LIBRARY)

# 与PREBUILT_SHARED_LIBRARY类似。
include $(PREBUILT_STATIC_LIBRARY)

# 指定生成目标的Android API level。
TARGET_PLATFORM := android-22

# 指定生成目标的CPU架构（armeabi，armeabi-v7a，arm64-v8a，x86，x86_64，mips，mips64或all）。
TARGET_ARCH_ABI := arm64-v8a

# 指定生成目标的API level和CPU架构。
TARGET_ABI := android-22-arm64-v8a
  • Module描述变量
# 获取当前文件路径。必须位于Android.mk开头，CLEAR_VARS无法重置该变量。
LOCAL_PATH := $(call my-dir)

# 生成文件名。（无需添加前缀lib和后缀.so或.a，编译系统会自动添加。)
LOCAL_MODULE := "foo"    # 如果指定生成动态库，则最终文件名为libfoo.so

# 生成文件名。（会覆盖LOCAL_MODULE所指定的文件名）
LOCAL_MODULE_FILENAME := libnewfoo    # 如果指定生成动态库，则最终文件名为libnewfoo.so

# 需要编译的源文件。（可使用相对路径或绝对路径，并使用斜杠"/"）
LOCAL_SRC_FILES := hello-jni.c hello.c

# C++扩展名。
LOCAL_CPP_EXTENSION := .cxx .cpp .cc

# C++依赖特性。（建议使用该变量替代使用LOCAL_CPPFLAGS ，因为该变量会在module需要时才加入特性，而 LOCAL_CPPFLAGS 则不管是否需要都加入。）
LOCAL_CPP_FEATURES := rtti exceptions

# 增加include文件的搜索路径。需在 LOCAL_CFLAGS 和 LOCAL_CPPFLAGS 之前设置该变量。（默认路径相对于于<ndk_root>。）
LOCAL_C_INCLUDES := sources/foo
# 可使用$(LOCAL_PATH)//foo。
LOCAL_C_INCLUDES := $(LOCAL_PATH)//foo

# 编译器flags。对C和C++源文件都有效。（可选。不改变optimization/debugging level的情况下，编译系统自动设置该变量）
LOCAL_CFLAGS += -I<path>
LOCAL_CPPFLAGS += -I<path>

# 生成文件是需要依赖的静态/动态库
LOCAL_STATIC_LIBRARIES
LOCAL_SHARED_LIBRARIES
LOCAL_WHOLE_STATIC_LIBRARIES

# 加载时链接其他库。生成动态库或可执行文件时可用。定义为静态库无效。（使用-l前缀）
LOCAL_LDLIBS := -lz    # links to /system/lib/libz.so

# 其他链接器
LOCAL_LDFLAGS += -fuse-ld=bfd    # -fuse-ld=gold is default

# --以下是其他不常用的变量--
LOCAL_ALLOW_UNDEFINED_SYMBOLS
LOCAL_ARM_MODE
LOCAL_ARM_NEON
LOCAL_DISABLE_NO_EXECUTE
LOCAL_DISABLE_RELRO
LOCAL_DISABLE_FORMAT_STRING_CHECKS
LOCAL_EXPORT_CFLAGS
LOCAL_EXPORT_CPPFLAGS
LOCAL_EXPORT_C_INCLUDES
LOCAL_EXPORT_LDFLAGS
LOCAL_EXPORT_LDLIBS
LOCAL_SHORT_COMMANDS
LOCAL_THIN_ARCHIVE
LOCAL_FILTER_ASM
  • NDK提供的函数宏
# 返回最后被include的makefile路径。一般是当前Android.mk所在文件夹。
my-dir

# 返回所有子文件夹的makefile文件路径
all-subdir-makefiles

# 返回当前makefile文件路径
this-makefile

# 返回父级makefile文件路径（即include当前makefile的那个makefile的路径）  
parent-makefile

# 返回祖父级makefile文件路径
grand-parent-makefile

# 查找并include一个module的Android.mk
import-module
$(call import-module,<name_of_module>)
